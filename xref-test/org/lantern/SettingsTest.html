56      @Test
57      public void testStartAtLogin() throws Exception {
58          if (!LanternConstants.LAUNCHD_PLIST.isFile()) {
59              log.info("No plist file - not installed or on different OS?");
60              return;
61          }
62          final Settings settings = LanternHub.settings();
63          final File temp = plist();
64          final File settingsFile = settingsFile();
65          FileUtils.copyFile(LanternConstants.LAUNCHD_PLIST, temp);
66          final String cur = FileUtils.readFileToString(temp, "UTF-8");
67          
68          assertTrue(cur.contains("<true/>") || cur.contains("<false/>"));
69          final SettingsIo ss = new SettingsIo(settingsFile);
70          final DefaultSettingsChangeImplementor implementor = 
71              new DefaultSettingsChangeImplementor(temp);
72          if (cur.contains("<true/>")) {
73              assertFalse(cur.contains("<false/>"));
74              //Configurator.setStartAtLogin(temp, false);
75              settings.setStartAtLogin(false);
76              implementor.setStartAtLogin(false);
77              ss.write(settings);
78              final String newFile = FileUtils.readFileToString(temp, "UTF-8");
79              assertTrue(newFile.contains("<false/>"));
80          } else if (cur.contains("<false/>")) {
81              assertFalse(cur.contains("<true/>"));
82              //Configurator.setStartAtLogin(temp, true);
83              settings.setStartAtLogin(true);
84              implementor.setStartAtLogin(true);
85              ss.write(settings);
86              final String newFile = FileUtils.readFileToString(temp, "UTF-8");
87              assertTrue(newFile.contains("<true/>"));
88          }
