44              assertTrue(test[0] + ',' + test[1], policy.accepts(cookie));
45          }
46  
47          String acceptFalse[][] = new String[][]{
48              {"name=value; domain=abc.example.org;", "http://www.example.org"},

50              {"name=value; domain=ww.example.org;", "http://www.example.org/"},
51              {"name=value; domain=org;", "http://www.example.org"},
52              {"name=value; domain=.org;", "http://www.example.org"},

66       * tests rules for whether a cookie can/should be 
67       * sent to a given request uri based on domain.
68       */ 
69      @Test 
70      public void testCanBeSent() throws Exception {
71  
72          StoredCookie cookie;
73  
74          
75          /* 
76           * domain matching
77           */
78          
79          
80          cookie = new StoredCookie("name", "value");
81          cookie.setDomain("bar.example.org");
82          cookie.setPath("/");
83          cookie.setSecure(false);
84  
85          // when the host only flag is set, only the exact 
86          // domain name can have the cookie. 
87          cookie.setHostOnly(true);
88          // this exact match will pass
89          assertTrue(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/")));
90          // all these non-exact matches will fail
91          assertFalse(CookieUtils.canBeSent(cookie, new URI("http://foo.bar.example.org/")));
92          assertFalse(CookieUtils.canBeSent(cookie, new URI("http://example.org/")));
93          assertFalse(CookieUtils.canBeSent(cookie, new URI("http://www.example.org/")));
94          assertFalse(CookieUtils.canBeSent(cookie, new URI("http://badbar.example.org/")));
95          assertFalse(CookieUtils.canBeSent(cookie, new URI("http://elsewhere.com/")));
96          
97          // all else being equals, if host only is false, exact matching
98          // is not required, only 'domain matching'
99          cookie.setHostOnly(false);
100         // these domain match
101         assertTrue(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/")));
102         assertTrue(CookieUtils.canBeSent(cookie, new URI("http://foo.bar.example.org/")));
103         // these should still fail.
104         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://example.org/")));
105         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://www.example.org/")));
106         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://badbar.example.org/")));
107         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://elsewhere.com/")));
108 
109         /*
110          * path matching 
111          */
112         cookie = new StoredCookie("name", "value");
113         cookie.setHostOnly(true);
114         cookie.setDomain("bar.example.org");
115         cookie.setSecure(false);
116         cookie.setPath("/foo");
117 
118         assertTrue(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/foo")));
119         assertTrue(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/foo/")));
120         assertTrue(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/foo/bar/baz")));
121         // these do not path match
122         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org")));
123         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/quux")));
124         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/foobad")));
125         
126         // exact prefix is required, so this means something slightly different
127         cookie.setPath("/foo/");
128         assertTrue(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/foo/")));
129         assertTrue(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/foo/bar/baz")));
130         // these do not path match
131         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/foo")));
132         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org")));
133         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/quux")));
134         assertFalse(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/foobad")));
135         
136         /*
137          * secure flag
138          */
139          cookie = new StoredCookie("name", "value");
140          cookie.setHostOnly(true);
141          cookie.setDomain("bar.example.org");
142          cookie.setPath("/");
143          cookie.setSecure(false);
144          
145          assertTrue(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/")));
146     
147          // currently we always say false when the secure flag has 
148          // somehow been set and it is visible to us because we 
149          // are never dealing with the contents of a secure end-to-end
150          // communication (those are invisible to us)
151          cookie.setSecure(true); 
152          assertFalse(CookieUtils.canBeSent(cookie, new URI("http://bar.example.org/")));
153     }

331     /**
332      * tests construction of uris with non-ascii 
333      * hostnames.
334      */
335     @Test
336     public void testSafeURI() throws Exception {
337         URI uri;
338         
339         String uriString = "http://foö.bar.org/quux";
340         
341         // first, assert the problem exists, otherwise we 
342         // may not need this rigamarole. Hostname with 
343         // non-ascii characters will yield a null host.
344         uri = new URI("http://foö.bar.org/quux");
345         assertTrue(uri.getHost() == null);
346 
347         // using makeSafeURI we should get the A-Name version 
348         // of the hostname.
349         uri = CookieUtils.makeSafeURI(uriString);
350         assertTrue(uri.getHost().equals("xn--fo-gka.bar.org"));
351         assertTrue(uri.getPath().equals("/quux"));
352         
353         // additional non-ascii characters in the url 
354         // outside the hostname should be the same.
355         uriString = "http://foö.bar.org/qöux";
356         uri = CookieUtils.makeSafeURI(uriString);
357         assertTrue(uri.getHost().equals("xn--fo-gka.bar.org"));
358         assertTrue(uri.getPath().equals("/qöux"));
359         
360         // preserves other junks, just modifies host
361         uriString = "https://flöke:flöp@foö.bar.org:9191/qöux?zbfö=zboe&foo#glöp";
362         uri = CookieUtils.makeSafeURI(uriString);
363         assertTrue(uri.getHost().equals("xn--fo-gka.bar.org"));
364         assertTrue(uri.getPath().equals("/qöux"));
365         assertTrue(uri.getUserInfo().equals("flöke:flöp"));
366         assertTrue(uri.getScheme().equals("https"));
367         assertTrue(uri.getPort() == 9191);
368         assertTrue(uri.getQuery().equals("zbfö=zboe&foo"));
369         assertTrue(uri.getFragment().equals("glöp"));
370         assertTrue(uri.toString().equals("https://flöke:flöp@xn--fo-gka.bar.org:9191/qöux?zbfö=zboe&foo#glöp"));
371 
372         // should be idempotent
373         uri = CookieUtils.makeSafeURI(uri.toString());
374         assertTrue(uri.getHost().equals("xn--fo-gka.bar.org"));
375         assertTrue(uri.getPath().equals("/qöux"));
376         assertTrue(uri.getUserInfo().equals("flöke:flöp"));
377         assertTrue(uri.getScheme().equals("https"));
378         assertTrue(uri.getPort() == 9191);
379         assertTrue(uri.getQuery().equals("zbfö=zboe&foo"));
380         assertTrue(uri.getFragment().equals("glöp"));
381         assertTrue(uri.toString().equals("https://flöke:flöp@xn--fo-gka.bar.org:9191/qöux?zbfö=zboe&foo#glöp"));
