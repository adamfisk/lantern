46      protected void buildBrowser() {
47          log.debug("Creating shell...");
48          
49          // This gets around a bug in XP/SWT/IE where SWT loads IE 7 even when
50          // IE 8 is on the user's system.
51          if (SystemUtils.IS_OS_WINDOWS_XP) {
52              System.setProperty("org.eclipse.swt.browser.IEVersion", "8000");
53          }
54          if (this.shell != null && !this.shell.isDisposed()) {
55              this.shell.forceActive();
56              return;
57          }
58          this.shell = new Shell(LanternHub.display());
59          final Image small = newImage("16on.png");
60          final Image medium = newImage("32on.png");
61          final Image large = newImage("128on.png");
62          final Image[] icons = new Image[]{small, medium, large};
63          shell.setImages(icons);
64          // this.shell = createShell(this.display);
65          shell.setText("Lantern Dashboard");
66          //this.shell.setSize(720, 540);
67          // shell.setFullScreen(true);
68  
69          final int minWidth = 970;
70          final int minHeight = 630;
71  
72          log.debug("Centering on screen...");
73          final Monitor primary = LanternHub.display().getPrimaryMonitor();
74          final Rectangle bounds = primary.getBounds();
75          final Rectangle rect = shell.getBounds();
76  
77          final int x = bounds.x + (bounds.width - rect.width) / 2;
78          final int y = bounds.y + (bounds.height - rect.height) / 2;
79  
80          shell.setLocation(x, y);
81          
82          log.debug("Creating new browser...");
83          final Browser browser = new Browser(shell, SWT.NONE);
84          log.debug("Running browser: {}", browser.getBrowserType());
85          browser.setSize(minWidth, minHeight);
86          //browser.setBounds(0, 0, 800, 600);
87          browser.setUrl("http://localhost:"+
88              LanternHub.settings().getApiPort());
89  
90          browser.addLocationListener(new LocationListener() {
91              @Override
92              public void changing(LocationEvent event) {
93                  try {
94                      final URL url = new URL(event.location);
95                      final String localAuthority = "localhost:" + // XXX aliases should work too
96                          LanternHub.settings().getApiPort();
97                      if (!url.getAuthority().equals(localAuthority)) {
98                          log.info("opening external browser to {}", event.location);
99                          event.doit = false;
100                         LanternUtils.browseUrl(event.location);
101                     }
102                 }
103                 catch (MalformedURLException e) {
104                     event.doit = false;
105                 }
106             }
107 
108             @Override
109             public void changed(LocationEvent event) {}
110         });
111 
112         // create a hidden browser to intercept external
113         // location references that should be openend
114         // in the system's native browser.
115         Shell hiddenShell = new Shell(LanternHub.display());
116         final Browser externalBrowser = new Browser(hiddenShell, SWT.NONE);
117 
118         externalBrowser.addLocationListener(new LocationListener() {
119             @Override
120             public void changing(LocationEvent event) {
121                 // launch external browser with link,
122                 // but don't actually go there.
123                 event.doit = false;
124                 LanternUtils.browseUrl(event.location);
125             }
126 
127             @Override
128             public void changed(LocationEvent event) {}
129         });
130 
131         browser.addOpenWindowListener(new OpenWindowListener() {
132             @Override
133             public void open(WindowEvent e) {
134                 e.browser = externalBrowser;
135             }
136         });
137 
138         shell.addListener (SWT.Close, new Listener () {
139             @Override
140             public void handleEvent(final Event event) {
141                 if (LanternHub.settings().isInitialSetupComplete()) {
142                     if (LanternHub.systemTray().isActive()) {
143                         browser.stop();
144                         browser.setUrl("about:blank");
145                     }
146                     else {
147                         final int style = SWT.APPLICATION_MODAL | SWT.ICON_INFORMATION | SWT.YES | SWT.NO;
148                         final MessageBox messageBox = new MessageBox (shell, style);
149                         messageBox.setText ("Quit Lantern?");
150                         final String msg = 
151                             "Quit Lantern?";
152                         messageBox.setMessage (msg);
153                         event.doit = messageBox.open () == SWT.YES;
154                         if (event.doit) {
155                             LanternHub.display().dispose();
156                             System.exit(0);
157                         }
158                     }
159                 } else {
160                     final int style = SWT.APPLICATION_MODAL | SWT.ICON_INFORMATION | SWT.YES | SWT.NO;
161                     final MessageBox messageBox = new MessageBox (shell, style);
162                     messageBox.setText ("Quit Lantern?");
163                     final String msg = 
164                         "Lantern setup has not been completed. Quit and set up later?";
165                     messageBox.setMessage (msg);
166                     event.doit = messageBox.open () == SWT.YES;
167                     if (event.doit) {
168                         LanternHub.display().dispose();
169                         System.exit(0);
170                     }
171                 }
172             }
173         });
174         shell.setLayout(new FillLayout());
175         Rectangle minSize = shell.computeTrim(0, 0, minWidth, minHeight); 
176         shell.setMinimumSize(minSize.width, minSize.height);
177         shell.pack();
178         shell.open();
179         shell.forceActive();
180         while (!shell.isDisposed()) {
181             if (!LanternHub.display().readAndDispatch())
182                 LanternHub.display().sleep();
183         }
184         hiddenShell.dispose();
185     }

187     private Image newImage(final String path) {
188         final String toUse;
189         final File path1 = new File(path);
190         if (path1.isFile()) {
191             toUse = path1.getAbsolutePath();
192         } else {
193             final File path2 = new File("install/common", path);
194             toUse = path2.getAbsolutePath();
195         }
196         return new Image(LanternHub.display(), toUse);

214      * Shows a dialog to the user asking a yes or no question.
215      * 
216      * @param title The title for the dialog.
217      * @param question The question to ask.
218      * @return <code>true</code> if the user answered yes, otherwise

222         return askQuestion(title, question, DEFAULT_QUESTION_FLAGS) == SWT.YES;
223     }

225     public int askQuestion(final String title, final String question, 
226         final int style) {
227         if (!LanternHub.settings().isUiEnabled()) {
228             log.info("MESSAGE BOX TITLE: "+title);
229             log.info("MESSAGE BOX MESSAGE: "+question);
230             return -1;
231         }
232         final AtomicInteger response = new AtomicInteger();
233         LanternHub.display().syncExec(new Runnable() {
234             @Override
235             public void run() {
236                 response.set(askQuestionOnThread(title, question, style));
237             }
238         });
239         log.info("Returned from sync exec");
240         return response.get();
241     }
242     
243     protected int askQuestionOnThread(final String title, 
244         final String question, final int style) {
245         log.info("Creating display...");
246         final Shell boxShell = new Shell(LanternHub.display());
247         log.info("Created display...");
248         final MessageBox messageBox = new MessageBox (boxShell, style);
249         messageBox.setText(title);
250         messageBox.setMessage(question);
251         return messageBox.open();
252     }
