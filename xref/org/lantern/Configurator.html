33      @Subscribe
34      public void onConnectivityEvent(final ConnectivityStatusChangeEvent event) {
35          final ConnectivityStatus status = event.getConnectivityStatus();
36          switch (status) {
37          case CONNECTED:
38              // Note that this call won't do anything on the first run if the
39              // setup screens have not completed -- proxy configuration in
40              // that case happens as a result of the setup complete API call.
41              // This call is necessary in subsequent cases, however, when the
42              // user has completed the setup screens, and Lantern simply 
43              // needs to start up in the correct proxy state.
44              configure();
45              break;
46          case CONNECTING:
47              break;
48          case DISCONNECTED:
49              break;
50          }
51      }
52      
53      public void configure() {
54          
55          LOG.info("Configuring...");
56          if (configured) {
57              LOG.info("Configure called multiple times?");
58              return;
59          }
60          configured = true;
61          reconfigure();
62      }

65      /**
66       * Copies our FireFox extension to the appropriate place.
67       * 
68       * @return The {@link File} for the final destination directory of the
69       * extension.
70       * @throws IOException If there's an error copying the extension.
71       */
72      public void copyFireFoxExtension() throws IOException {
73          LOG.info("Copying FireFox extension");
74          final File dir = getExtensionDir();
75          if (!dir.isDirectory()) {
76              LOG.info("Making FireFox extension directory...");
77              // NOTE: This likely means the user does not have FireFox. We copy
78              // the extension here anyway in case the user ever installs 
79              // FireFox in the future.
80              if (!dir.mkdirs()) {
81                  LOG.error("Could not create ext dir: "+dir);
82                  throw new IOException("Could not create ext dir: "+dir);
83              }
84          }
85          final String extName = "lantern@getlantern.org";
86          final File dest = new File(dir, extName);
87          final File ffDir = new File("firefox/"+extName);
88          if (dest.exists() && !FileUtils.isFileNewer(ffDir, dest)) {
89              LOG.info("Extension appears to already exist and ours is not newer");
90              return;
91          }
92          if (!ffDir.isDirectory()) {
93              LOG.error("No extension directory found at {}", ffDir);
94              throw new IOException("Could not find extension?");
95          }
96          FileUtils.copyDirectoryToDirectory(ffDir, dir);
97          LOG.info("Copied FireFox extension from {} to {}", ffDir, dir);
98      }

113     public void reconfigure() {
114         if (!LanternUtils.isConfigured()) {
115             System.out.println("GOOGLE ACCOUNT NOT CONFIGURED");
116             return;
117         }
118         final File git = new File(".git");
119         if (git.isDirectory() && !LanternHub.settings().isGetMode()) {
120             LOG.info("Running from repository...not auto-configuring proxy.");
121             return;
122         }
123         
124         if (LanternUtils.shouldProxy() &&
125             (!LanternHub.settings().isUiEnabled() || LanternHub.settings().isInitialSetupComplete())) {
126             LOG.info("Auto-configuring proxy...");
127             boolean finished = false;
128             while (!finished) {
129                 try {
130                     Proxifier.startProxying();
131                     finished = true;
132                 } catch (Proxifier.ProxyConfigurationError e) {
133                     if (LanternHub.settings().isUiEnabled()) {
134                          // XXX I18n / copy 
135                          final String question = "Failed to set Lantern as the system proxy.\n\n" +
136                             "If you cancel, Lantern will not be used to handle " +
137                             "your web traffic unless you manually configure your proxy settings.\n\n" +
138                             "Try again?";
139                         final int response = 
140                             LanternHub.dashboard().askQuestion("Proxy Settings", question,
141                                 SWT.APPLICATION_MODAL | SWT.ICON_INFORMATION | SWT.RETRY | SWT.CANCEL);
142                         if (response == SWT.CANCEL) {
143                             finished = true;
144                         }
145                     }
146                     else {
147                         LOG.error("Failed to set lantern as the system proxy: {}", e);
148                         finished = true;
149                     }
150                 }
151             }
152         } else {
153             LOG.info("Not auto-configuring proxy.");
154         }
155     }

162      * Installs the FireFox config file on startup. 
163      */
164     private void copyFirefoxConfig() {
165         final File ff;
166         if (SystemUtils.IS_OS_WINDOWS) {
167             ff = new File(System.getenv("ProgramFiles"), "Mozilla Firefox");
168         } else if (SystemUtils.IS_OS_MAC_OSX) {
169             ff = new File("/Applications/Firefox.app//Contents/MacOS");
170         } else {
171             LOG.info("Not sure where to copy FireFox config on Linux");
172             return;
173         }
174         final File pref = new File(new File(ff, "defaults"), "pref");
175         LOG.info("Prefs dir: {}", pref);
176         if (ff.isDirectory() && !pref.isDirectory()) {
177             LOG.error("No directory at: {}", pref);
178         }
179         final File config = new File("all-lantern.js");
180         
181         if (!config.isFile()) {
182             LOG.error("NO CONFIG FILE AT {}", config);
183         }
184         else {
185             try {
186                 FileUtils.copyFileToDirectory(config, pref);
187                 final File installedConfig = new File(pref, config.getName());
188                 installedConfig.deleteOnExit();
189             } catch (final IOException e) {
190                 LOG.error("Could not copy config file?", e);
191             }
192         }
193     }
