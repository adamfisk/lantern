22  /**
23   * Handles connections from the local proxy to external proxies, relaying
24   * data back to the original channel to the browser.
25   */
26  public class OutboundHandler extends SimpleChannelUpstreamHandler {
27  
28      private final Logger log = LoggerFactory.getLogger(getClass());
29      
30      private static final long CHUNK_SIZE = 1024 * 1024 * 10 - (2 * 1024);
31      
32      private final Channel browserToProxyChannel;
33  
34      private final Queue<HttpRequest> httpRequests;
35  
36      public OutboundHandler(final Channel browserToProxyChannel,
37          final Queue<HttpRequest> httpRequests) {
38          this.browserToProxyChannel = browserToProxyChannel;
39          this.httpRequests = httpRequests;
40      }
41      
42      @Override
43      public void messageReceived(final ChannelHandlerContext ctx, 
44          final MessageEvent e) {
45          final Object msg = e.getMessage();
46          
47          if (msg instanceof HttpChunk) {
48              final HttpChunk chunk = (HttpChunk) msg;
49              
50              if (chunk.isLast()) {
51                  log.info("GOT LAST CHUNK");
52              }
53              browserToProxyChannel.write(chunk);
54          } else {
55              log.info("Got message on outbound handler: {}", msg);
56              // There should always be a one-to-one relationship between
57              // requests and responses, so we want to pop a request off the
58              // queue for every response we get in. This is only really
59              // needed so we have all the appropriate request values for
60              // making additional requests to handle 206 partial responses.
61              final HttpRequest request = httpRequests.remove();
62              //final ChannelBuffer msg = (ChannelBuffer) e.getMessage();
63              //if (msg instanceof HttpResponse) {
64              final HttpResponse response = (HttpResponse) msg;
65              final int code = response.getStatus().getCode();
66              if (code != 206) {
67                  log.info("No 206. Writing whole response");
68                  browserToProxyChannel.write(response);
69              } else {
70                  
71                  
72                  // We just grab this before queuing the request because
73                  // we're about to remove it.
74                  final String cr = 
75                      response.getHeader(HttpHeaders.Names.CONTENT_RANGE);
76                  final long cl = parseFullContentLength(cr);
77                  
78                  if (isFirstChunk(cr)) {
79                      // If this is the *first* partial response to this 
80                      // request, we need to build a new HTTP response as if 
81                      // it were a normal, non-partial 200 OK. We need to 
82                      // make sure not to do this for *every* 206 response, 
83                      // however.
84                      response.setStatus(HttpResponseStatus.OK);
85                      
86                      log.info("Setting Content Length to: "+cl+" from "+cr);
87                      response.setHeader(HttpHeaders.Names.CONTENT_LENGTH, cl);
88                      response.removeHeader(HttpHeaders.Names.CONTENT_RANGE);
89                      browserToProxyChannel.write(response);
90                  } else {
91                      // We need to grab the body of the partial response
92                      // and return it as an HTTP chunk.
93                      final HttpChunk chunk = 
94                          new DefaultHttpChunk(response.getContent());
95                      browserToProxyChannel.write(chunk);
96                  }
97                  
98                  // Spin up additional requests on a new thread.
99                  requestRange(request, response, cr, cl, ctx.getChannel());
100             }
101         }
102     }

105     private boolean isFirstChunk(final String contentRange) {
106         return contentRange.trim().startsWith("bytes 0-");
107     }
108 
109     private long parseFullContentLength(final String contentRange) {
110         final String fullLength = 
111             StringUtils.substringAfterLast(contentRange, "/");
112         return Long.parseLong(fullLength);
113     }
114 
115     private void requestRange(final HttpRequest request, 
116         final HttpResponse response, final String contentRange, 
117         final long fullContentLength, final Channel channel) {
118         log.info("Queuing request based on Content-Range: {}", contentRange);
119         // Note we don't need to thread this since it's all asynchronous 
120         // anyway.
121         final String body = 
122             StringUtils.substringAfter(contentRange, "bytes ");
123         if (StringUtils.isBlank(body)) {
124             log.error("Blank bytes body: "+contentRange);
125             return;
126         }
127         final long contentLength = HttpHeaders.getContentLength(response);
128         final String startPlus = StringUtils.substringAfter(body, "-");
129         final String startString = StringUtils.substringBefore(startPlus, "/");
130         final long start = Long.parseLong(startString) + 1;
131         
132         // This means the last response provided the final range, so we don't
133         // want to request another one.
134         if (start == fullContentLength) {
135             log.info("Received full length...not requesting new range");
136             return;
137         }
138         final long end;
139         if (contentLength - start > CHUNK_SIZE) {
140             end = start + CHUNK_SIZE;
141         } else {
142             end = fullContentLength - 1;
143         }
144         request.setHeader(HttpHeaders.Names.RANGE, "bytes="+start+"-"+end);
145         writeRequest(request, channel);
146     }
147     
148     /**
149      * Helper method that ensures all written requests are properly recorded.
150      * 
151      * @param request The request.
152      */
153     private void writeRequest(final HttpRequest request, final Channel channel) {
154         this.httpRequests.add(request);
155         log.info("Writing request: {}", request);
156         channel.write(request);
157     }

160     public void channelClosed(final ChannelHandlerContext ctx, 
161         final ChannelStateEvent e) throws Exception {
162         log.info("Channel to external proxy closed");
163         ProxyUtils.closeOnFlush(browserToProxyChannel);
164     }
