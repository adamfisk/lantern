554     /** 
555      * copy properties annotated with the given jsonView class 
556      * from this Settings object to the Settings object given. 
557      * 
558      * copy is shallow!
559      */
560     public void copyView(Settings into, Class<?> selector)
561         throws IllegalAccessException, IllegalArgumentException, 
562                InvocationTargetException, NoSuchMethodException {
563         for (final Method method : Settings.class.getMethods()) {
564             if (method.isAnnotationPresent(JsonView.class)) {
565                 final JsonView v = method.getAnnotation(JsonView.class);
566                 for (final Class<?> c : v.value()) {
567                     if (c == selector) {
568                         // method is annotated with selected JsonView
569                         // try to transfer property.
570                         final String propertyName = LanternUtils.methodNameToProperty(method.getName()); 
571                         if (propertyName != null) {
572                             PropertyUtils.setSimpleProperty(into, propertyName,
573                                 PropertyUtils.getSimpleProperty(this, propertyName));
574                             log.debug("copied setting {}", propertyName);
575                         }
576                         else {
577                             log.error("Skipping copy of annotated but unbeanish method \"{}\": can't determine prop name", method.getName());
578                         }
579                     }
580                 }
581             }
582         }
