22      /**
23       * @return true if and only if it is legitimate to send the 
24       * StoredCookie given to the requestUri given according to 
25       * RFC 6265 Section 5.4.1
26       */
27      public static boolean canBeSent(final StoredCookie cookie, final URI requestUri) {
28  
29          final String canonicalHost = canonicalizeHost(requestUri.getHost());
30          final String canonicalDomain = canonicalizeHost(cookie.getDomain());
31  
32          // RFC 6265 requires ALL of the following be true: 
33          
34          // *  Either:
35          // 
36          //       The cookie's host-only-flag is true and the canonicalized
37          //       request-host is identical to the cookie's domain.
38          // 
39          if (cookie.isHostOnly()) {
40              if (!canonicalHost.equals(canonicalDomain)) {
41                  return false;
42              }
43          }
44          //    Or:
45          // 
46          //       The cookie's host-only-flag is false and the canonicalized
47          //       request-host domain-matches the cookie's domain.
48          //
49          else {
50              if (!domainMatches(canonicalDomain, canonicalHost)) {
51                  return false;
52              }
53          }
54  
55          // 
56          // *  The request-uri's path path-matches the cookie's path.
57          //
58          if (!pathMatches(cookie.getPath(), requestUri.getPath())) {
59              return false;
60          }
61  
62          // 
63          // *  If the cookie's secure-only-flag is true, then the request-
64          //    uri's scheme must denote a "secure" protocol (as defined by
65          //    the user agent).
66          // 
67          if (cookie.isSecure()) {
68              // [ possibly there is something else to think about here, 
69              // but in general, if we're able to consider it, 
70              // we're doing so on an "insecure" channel. So anything
71              // that was marked as secure by whatever means should not be
72              // sent by lantern. Actual secure cookies will be invisible to us
73              // end-to-end. ] 
74              return false; 
75          }
76  
77          // *  If the cookie's http-only-flag is true, then exclude the
78          //    cookie if the cookie-string is being generated for a "non-
79          //    HTTP" API (as defined by the user agent).
80          
81          // [ not sure there is anything to consider here ]
82  
83          return true; 
84      
85      }

87      /**
88       * @return true if and only if the cookie Domain setting 
89       * matches the given hostname according to RFC 6265 Section 5.1.3
90       *
91       * @param cookieDomain canonical cookie domain
92       * @param hostname canonical hostname to path match against cookieDomain
93       */
94      public static boolean domainMatches(final String cookieDomain, final String hostname) {
95          
96          // A string domain-matches a given domain string if at least one of the
97          // following conditions hold:
98          // 
99          // o  The domain string and the string are identical.  (Note that both
100         //    the domain string and the string will have been canonicalized to
101         //    lower case at this point.)        
102         if (cookieDomain.equals(hostname)) {
103             return true;
104         }
105 
106         // o  All of the following conditions hold:
107         // 
108         //    *  The domain string is a suffix of the string.
109         // 
110         //    *  The last character of the string that is not included in the
111         //       domain string is a %x2E (".") character.
112         // 
113         //    *  The string is a host name (i.e., not an IP address).
114         if (!hostname.endsWith(cookieDomain) ||
115             hostname.charAt(hostname.length() - cookieDomain.length() - 1) != '.' ||
116             InetAddresses.isInetAddress(hostname)) {
117             return false;
118         }
119 
120         return true;    
121     }

123     /**
124      * return a canonical form of the hostname given 
125      * suitable for matching and domain matching in 
126      * RFC6265 as described in Section 5.1.2
127      *
128      */
129     public static String canonicalizeHost(final String host) {
130 
131         // TODO RFC6265 Section 5.1.2 
132         //        
133         // A canonicalized host name is the string generated by the following
134         // algorithm:
135         // 
136         //   1.  Convert the host name to a sequence of individual domain name
137         //       labels.
138         // 
139         //   2.  Convert each label that is not a Non-Reserved LDH (NR-LDH) label,
140         //       to an A-label (see Section 2.3.2.1 of [RFC5890] for the former
141         //       and latter), or to a "punycode label" (a label resulting from the
142         //       "ToASCII" conversion in Section 4 of [RFC3490]), as appropriate
143         //       (see Section 6.3 of this specification).
144         // 
145         //   3.  Concatenate the resulting labels, separated by a %x2E (".")
146         //       character.
147 
148         // [ java.net.IDN.toASCII supports full hostnames and RFC3490 ]
149         return IDN.toASCII(host).toLowerCase();
150     }

152     /**
153      * @return true if and only if the Path setting of the given cookie
154      * matches the given request path according to RFC 6265 Section 5.1.4
155      *
156      */
157     public static boolean pathMatches(final String cookiePath, final String requestPath) {
158         // A request-path path-matches a given cookie-path if at least one of
159         // the following conditions holds:
160 
161         // o  The cookie-path and the request-path are identical.
162         // 
163         // [ also considering empty requestPath to be equal to "/" -- it is required to be treated as such 
164         //   elswhere in standards, but our pipeline does not enforce this literally ]
165         if (cookiePath.equals(requestPath) || (cookiePath.equals("/") && requestPath.equals(""))) {
166             return true; 
167         }
168 
169         // o  The cookie-path is a prefix of the request-path, and the last
170         //    character of the cookie-path is %x2F ("/").
171         //
172         // o  The cookie-path is a prefix of the request-path, and the first
173         //    character of the request-path that is not included in the cookie-
174         //    path is a %x2F ("/") character.            
175         if (requestPath.startsWith(cookiePath) && 
176             (cookiePath.endsWith("/") || requestPath.charAt(cookiePath.length()) == '/')) {
177             return true;
178         }
179 
180         return false;
181     }

189         // consult guava's implementation of the public suffix list
190         try {
191             return InternetDomainName.fromLenient(domain).isPublicSuffix();
192         }
193         catch (IllegalArgumentException e) {
194             // not a valid hostname

199     public static String normalizedSetCookieDomain(final String domain) {
200         
201         // RFC 6265 Section 5.2.3
202 
203         // If the attribute-value is empty, the behavior is undefined.  However,
204         // the user agent SHOULD ignore the cookie-av entirely.
205         //
206         if (domain == null || domain.equals("")) {
207             return null;
208         }
209         
210         // If the first character of the attribute-value string is %x2E ("."):
211         // 
212         //    Let cookie-domain be the attribute-value without the leading %x2E
213         //    (".") character.
214         // 
215         else if (domain.startsWith(".")) {
216             return domain.substring(1).toLowerCase();
217         }
218 
219         //
220         // Otherwise:
221         // 
222         //    Let cookie-domain be the entire attribute-value.
223         // 
224         // Convert the cookie-domain to lower case.
225 
226         else {
227             return domain.toLowerCase();
228         }
229     }

231     public static String normalizedSetCookiePath(final String path, final URI originUri) {
232         // RFC 6265 Section 5.2.4
233 
234         // If the attribute-value is empty or if the first character of the
235         // attribute-value is not %x2F ("/"):
236         // 
237         //    Let cookie-path be the default-path.
238         if (path == null || path.equals("") || path.charAt(0) != '/') {
239             return defaultSetCookiePath(originUri);
240         }
241 
242         // 
243         // Otherwise:
244         // 
245         //    Let cookie-path be the attribute-value.
246         // 
247         return path;
248     }

251     /**
252      * Compute the default Path for a cookie sent from the given originUri 
253      * according to RFC 6265 Section 5.1.4
254      *
255      */
256     public static String defaultSetCookiePath(final URI requestUri) {
257 
258         // 1.  Let uri-path be the path portion of the request-uri if such a
259         //     portion exists (and empty otherwise).  For example, if the
260         //     request-uri contains just a path (and optional query string),
261         //     then the uri-path is that path (without the %x3F ("?") character
262         //     or query string), and if the request-uri contains a full
263         //     absoluteURI, the uri-path is the path component of that URI.
264         final String uriPath = requestUri.getPath();
265 
266         // 2.  If the uri-path is empty or if the first character of the uri-
267         //     path is not a %x2F ("/") character, output %x2F ("/") and skip
268         //     the remaining steps.
269         if (uriPath == null || uriPath.equals("") || uriPath.charAt(0) != '/') {
270             return "/";
271         }
272 
273         // 3.  If the uri-path contains no more than one %x2F ("/") character,
274         //     output %x2F ("/") and skip the remaining step.
275         int lastSlash = uriPath.lastIndexOf('/');
276         if (lastSlash <= 0) { // already verified it starts with "/" so this indicates 1 / only. 
277             return "/";
278         }
279 
280         // 4.  Output the characters of the uri-path from the first character up
281         //     to, but not including, the right-most %x2F ("/").
282         return uriPath.substring(0, lastSlash);        
283     }

285     /**
286      * create a URI that may be a URL with unicode in the hostname. 
287      * If so, first convert the hostname to A-Labels via RFC 3490.
288      * Works around an issue with constructing URIs with urls 
289      * containing unicode hostnames.
290      */
291     public static URI makeSafeURI(final String uri) throws URISyntaxException {
292         try {
293             final URL url = new URL(uri);
294             final String safeHost = IDN.toASCII(url.getHost());
295             if (safeHost.equals(url.getHost())) {
296                 return new URI(uri);
297             }
298             else {
299                 // construct a URL String with the IDN.toASCII version 
300                 // of the host (A-Labels)
301                 
302                 // there is no general way to re-construct or modify 
303                 // a URL or URI other than by completely re-represeting it a
304                 // String and calling the single String constructor.
305                 // All other constructors are mainly intended to test the 
306                 // patience of developers. 
307                 final StringBuffer newURL = new StringBuffer();
308                 newURL.append(url.getProtocol());
309                 newURL.append("://");
310                 if (url.getUserInfo() != null) {
311                     newURL.append(url.getUserInfo());
312                     newURL.append("@"); 
313                 }
314                 newURL.append(safeHost);
315                 if (url.getPort() != -1) {
316                     newURL.append(":" + url.getPort());
317                 }
318                 newURL.append(url.getFile());
319                 if (url.getRef() != null) {
320                     newURL.append("#");
321                     newURL.append(url.getRef());
322                 }
323                 return new URI(newURL.toString());
324             }
325         }
326         catch (MalformedURLException e) {
327             // not a url.. who knows maybe we're making a request to 
328             // an isbn number, just do our best...
329             return new URI(uri);
330         }
331     }

339         final HttpVersion ver = toCopy.getProtocolVersion();
340         final HttpVersion verCopy = new HttpVersion(ver.getText(), ver.isKeepAliveDefault());
341         final DefaultHttpRequest request = new DefaultHttpRequest(verCopy, toCopy.getMethod(), toCopy.getUri());

345             request.setHeader(headerName, toCopy.getHeaders(headerName));
346         }

351     public static class RFC6265SetCookieFilter implements CookieFilter {
352         
353         String canonicalHostname;
354         
355         public RFC6265SetCookieFilter(HttpRequest context) {
356             try {
357                 final URI requestUri = makeSafeURI(context.getUri());
358                 canonicalHostname = canonicalizeHost(requestUri.getHost());
359             }
360             catch (URISyntaxException e) {
361                 canonicalHostname = null;
362             }
363         }
364 
365         /**
366          * @return true if and only if the Cookie given can be legitimately set on a request
367          * to the URI of the request context according to the Cookie's domain setting
368          * and RFC 6265 Section 5.3
369          * 
370          * eg you cannot set a Cookie on facebook.com when making a 
371          * request to http://malicious.example.com/ etc.
372          *
373          */
374         @Override
375         public boolean accepts(final Cookie cookie) {
376 
377             //
378             // Rejection steps from RFC 6265 Sections 5.3.5-5.3.6
379             // 
380             // netty / lantern does not exactly follow the parsing and processing model 
381             // of RFC 6265, which requires rejecting cookies on certain bases during 
382             // parsing.  Instead, those considerations are collected here and make
383             // certain equivalent rejections.
384             // 
385             // Other normalization is expected to have been carried out, minus 
386             // the immediate rejection steps, @see StoredCookie.fromSetCookie
387             // 
388             if (!(cookie instanceof StoredCookie) || canonicalHostname == null) {
389                 return false; 
390             }
391 
392             final StoredCookie storedCookie = (StoredCookie) cookie;
393 
394             final String canonicalDomain = canonicalizeHost(storedCookie.getDomain());
395 
396             // if the cookie is host-only, only exact matches are permitted 
397             // between the host. Interestingly, it is allowed to be a public suffix 
398             // in this exact match case as, strangely, some are also websites. X_x
399             if (storedCookie.isHostOnly()) {
400                 return canonicalDomain.equals(canonicalHostname);
401             }
402 
403             // if it is not host-only, it must domain match.
404             if (!domainMatches(canonicalDomain, canonicalHostname)) {
405                 return false;
406             }
407 
408             // cannot be a public suffix unless host-only.
409             if (isPublicSuffix(canonicalDomain)) {
410                 return false;
411             }
412 
413             return true;
414         }
415                 
416     }
