107     private static void launch(final String... args) {
108         LOG.info("Starting Lantern...");
109         configureCipherSuites();
110         
111         // first apply any command line settings
112         final Options options = new Options();
113         options.addOption(null, OPTION_DISABLE_UI, false,
114                           "run without a graphical user interface.");
115         options.addOption(null, OPTION_API_PORT, true,
116             "the port to run the API server on.");
117         options.addOption(null, OPTION_PUBLIC_API, false,
118             "make the API server publicly accessible on non-localhost.");
119         options.addOption(null, OPTION_HELP, false,
120                           "display command line help");
121         options.addOption(null, OPTION_LAUNCHD, false,
122             "running from launchd - not normally called from command line");
123         options.addOption(null, OPTION_DISABLE_KEYCHAIN, false, 
124             "disable use of system keychain and ask for local password");
125         options.addOption(null, OPTION_PASSWORD_FILE, true, 
126             "read local password from the file specified");
127         
128         options.addOption(null, OPTION_TRUSTED_PEERS, false,
129             "disable use of trusted peer-to-peer connections for proxies.");
130         options.addOption(null, OPTION_ANON_PEERS, false,
131             "disable use of anonymous peer-to-peer connections for proxies.");
132         options.addOption(null, OPTION_LAE, false,
133             "disable use of app engine proxies.");
134         options.addOption(null, OPTION_CENTRAL, false,
135             "disable use of centralized proxies.");
136         options.addOption(null, OPTION_UDP, false,
137             "disable UDP for peer-to-peer connections.");
138         options.addOption(null, OPTION_TCP, false,
139             "disable TCP for peer-to-peer connections.");
140         options.addOption(null, OPTION_USER, true,
141             "Google user name -- WARNING INSECURE - ONLY USE THIS FOR TESTING!");
142         options.addOption(null, OPTION_PASS, true,
143             "Google password -- WARNING INSECURE - ONLY USE THIS FOR TESTING!");
144         options.addOption(null, OPTION_GET, false, "Force running in get mode");
145         options.addOption(null, OPTION_GIVE, false, "Force running in give mode");
146         
147         final CommandLineParser parser = new PosixParser();
148         final CommandLine cmd;
149         try {
150             cmd = parser.parse(options, args);
151             if (cmd.getArgs().length > 0) {
152                 throw new UnrecognizedOptionException("Extra arguments were provided");
153             }
154         }
155         catch (final ParseException e) {
156             printHelp(options, e.getMessage()+" args: "+Arrays.asList(args));
157             return;
158         }
159         
160         if (cmd.hasOption(OPTION_HELP)) {
161             printHelp(options, null);
162             return;
163         }
164         final Settings set = LanternHub.settings();
165         set.setUseTrustedPeers(parseOptionDefaultTrue(cmd, OPTION_TRUSTED_PEERS));
166         set.setUseAnonymousPeers(parseOptionDefaultTrue(cmd, OPTION_ANON_PEERS));
167         set.setUseLaeProxies(parseOptionDefaultTrue(cmd, OPTION_LAE));
168         set.setUseCentralProxies(parseOptionDefaultTrue(cmd, OPTION_CENTRAL));
169         
170         IceConfig.setTcp(parseOptionDefaultTrue(cmd, OPTION_TCP));
171         IceConfig.setUdp(parseOptionDefaultTrue(cmd, OPTION_UDP));
172         
173         if (cmd.hasOption(OPTION_USER)) {
174             set.setCommandLineEmail(cmd.getOptionValue(OPTION_USER));
175         }
176         if (cmd.hasOption(OPTION_PASS)) {
177             set.setCommandLinePassword(cmd.getOptionValue(OPTION_PASS));
178         }
179         if (cmd.hasOption(OPTION_DISABLE_UI)) {
180             LOG.info("Disabling UI");
181             set.setUiEnabled(false);
182         }
183         else {
184             set.setUiEnabled(true);
185         }
186         
187         /* option to disable use of keychains in local privacy */
188         if (cmd.hasOption(OPTION_DISABLE_KEYCHAIN)) {
189             LOG.info("Disabling use of system keychains");
190             set.setKeychainEnabled(false);
191         }
192         else {
193             set.setKeychainEnabled(true);
194         }
195         
196         if (cmd.hasOption(OPTION_PASSWORD_FILE)) {
197             loadLocalPasswordFile(cmd.getOptionValue(OPTION_PASSWORD_FILE));
198         }
199         
200         if (cmd.hasOption(OPTION_PUBLIC_API)) {
201             set.setBindToLocalhost(false);
202         }
203         if (cmd.hasOption(OPTION_API_PORT)) {
204             final String portStr = 
205                 cmd.getOptionValue(OPTION_API_PORT);
206             LOG.info("Using command-line port: "+portStr);
207             final int port = Integer.parseInt(portStr);
208             set.setApiPort(port);
209         } else {
210             LOG.info("Using random port...");
211             set.setApiPort(LanternUtils.randomPort());
212         }
213         LOG.info("Running API on port: {}", set.getApiPort());
214 
215         if (cmd.hasOption(OPTION_LAUNCHD)) {
216             LOG.info("Running from launchd or launchd set on command line");
217             set.setLaunchd(true);
218         } else {
219             set.setLaunchd(false);
220         }
221         
222         final Display display;
223         if (set.isUiEnabled()) {
224             // We initialize this super early in case there are any errors 
225             // during startup we have to display to the user.
226             Display.setAppName("Lantern");
227             display = LanternHub.display();
228             // Also, We need the system tray to listen for events early on.
229             LanternHub.systemTray().createTray();
230         }
231         else {
232             display = null;
233         }
234         
235         loadSettings();
236         
237         if (cmd.hasOption(OPTION_GIVE)) {
238             LanternHub.settings().setGetMode(false);
239         } else if (cmd.hasOption(OPTION_GET)) {
240             LanternHub.settings().setGetMode(true);
241         }
242         
243         gnomeAutoStart();
244         
245         // Use our stored STUN servers if available.
246         final Collection<String> stunServers = 
247             LanternHub.settings().getStunServers();
248         if (stunServers != null && !stunServers.isEmpty()) {
249             LOG.info("Using stored STUN servers: {}", stunServers);
250             StunServerRepository.setStunServers(toSocketAddresses(stunServers));
251         }
252         if (LanternUtils.hasNetworkConnection()) {
253             LOG.info("Got internet...");
254             launchWithOrWithoutUi();
255         } else {
256             // If we're running on startup, it's quite likely we just haven't
257             // connected to the internet yet. Let's wait for an internet
258             // connection and then start Lantern.
259             if (LanternHub.settings().isLaunchd() || !LanternHub.settings().isUiEnabled()) {
260                 LOG.info("Waiting for internet connection...");
261                 LanternUtils.waitForInternet();
262                 launchWithOrWithoutUi();
263             }
264             // If setup is complete and we're not running on startup, open
265             // the dashboard.
266             else if (LanternHub.settings().isInitialSetupComplete()) {
267                 LanternHub.jettyLauncher().openBrowserWhenReady();
268                 // Wait for an internet connection before starting the XMPP
269                 // connection.
270                 LOG.info("Waiting for internet connection...");
271                 LanternUtils.waitForInternet();
272                 launchWithOrWithoutUi();
273             } else {
274                 // If we haven't configured Lantern and don't have an internet
275                 // connection, the problem is that we can't verify the user's
276                 // user name and password when they try to login, so we just
277                 // let them know we can't start Lantern until they have a 
278                 // connection.
279                 // TODO: i18n
280                 final String msg = 
281                     "We're sorry, but you cannot configure Lantern without " +
282                     "an active connection to the internet. Please try again " +
283                     "when you have an internet connection.";
284                 LanternHub.dashboard().showMessage("No Internet", msg);
285                 System.exit(0);
286             }
287         }
288 
289         
290         // This is necessary to keep the tray/menu item up in the case
291         // where we're not launching a browser.
292         if (display != null) {
293             while (!display.isDisposed ()) {
294                 if (!display.readAndDispatch ()) display.sleep ();
295             }
296         }
297     }

349     private static void configureCipherSuites() {
350         if (!LanternUtils.isUnlimitedKeyStrength()) {
351             if (!SystemUtils.IS_OS_WINDOWS_VISTA) {
352                 LOG.error("No policy files on non-Vista machine!!");
353             }
354             LOG.info("Reverting to weaker ciphers on Vista");
355             IceConfig.setCipherSuites(new String[] {
356                 "TLS_DHE_RSA_WITH_AES_128_CBC_SHA"
357             });
358         } else {
359             // Note the following just sets what cipher suite the server 
360             // side selects. DHE is for perfect forward secrecy.
361             
362             // CBC mitigates the BEAST attack. We also include 128 because 
363             // we never have enough permissions to copy the unlimited  
364             // strength policy files on Vista, so we have to revert back
365             // to 128.
366             IceConfig.setCipherSuites(new String[] {
367                 "TLS_DHE_RSA_WITH_AES_256_CBC_SHA",
368                 "TLS_DHE_RSA_WITH_AES_128_CBC_SHA"
369             });
370         }
371     }

374     private static Collection<InetSocketAddress> toSocketAddresses(
375         final Collection<String> stunServers) {
376         final Collection<InetSocketAddress> isas = 
377             new HashSet<InetSocketAddress>();
378         for (final String server : stunServers) {
379             final String host = StringUtils.substringBefore(server, ":");
380             final String port = StringUtils.substringAfter(server, ":");
381             isas.add(new InetSocketAddress(host, Integer.parseInt(port)));
382         }
383         return isas;
384     }

407     // TODO: We want to make it such that once this method returns the
408     // settings are always fully loaded or the entire app has aborted, 
409     // including prompting the user for password.
410     private static void loadSettings() {
411         LanternHub.resetSettings(true);
412         if (LanternHub.settings().getSettings().getState() == SettingsState.State.CORRUPTED) {
413             try {
414                 // current behavior is automatic reset of all local data / ciphers
415                 // immediately.  This behavior could be deferred until later or handled
416                 // in some other way.
417                 LOG.warn("Destroying corrupt settings...");
418                 LanternHub.destructiveFullReset();
419             }
420             catch (IOException e) {
421                 LOG.error("Failed to reset corrupt settings: {}", e);
422                 System.exit(1);
423             }
424             // still corrupt?
425             if (LanternHub.settings().getSettings().getState() == SettingsState.State.CORRUPTED) {
426                 LOG.error("Failed to reset corrupt settings.");
427                 System.exit(1);
428             }
429             else {
430                 LOG.info("Settings have been reset.");
431             }
432         }
433         
434         // if there is no UI and the settings are locked, we need to grab the password on 
435         // the command line or else quit.
436         if (!LanternHub.settings().isUiEnabled() && 
437             LanternHub.settings().getSettings().getState() == SettingsState.State.LOCKED) {
438             if (!askToUnlockSettingsCLI()) {
439                 LOG.error("Unable to unlock settings.");
440                 System.exit(1);
441             }
442         }
443         
444         LOG.info("Settings state is {}", LanternHub.settings().getSettings().getState());
445     }

447     private static boolean askToUnlockSettingsCLI() {
448         if (!LanternHub.localCipherProvider().requiresAdditionalUserInput()) {
449             LOG.info("Local cipher does not require a password.");
450             return true;
451         }
452         while(true) {
453             char [] pw = null; 
454             try {
455                 pw = readSettingsPasswordCLI();
456                 return unlockSettingsWithPassword(pw);
457             }
458             catch (final InvalidKeyException e) {
459                 System.out.println("Password was incorrect, try again."); // XXX i18n
460             }
461             catch (final GeneralSecurityException e) {
462                 LOG.error("Error unlocking settings: {}", e);
463             }
464             catch (final IOException e) {
465                 LOG.error("Erorr unlocking settings: {}", e);
466             }
467             finally {
468                 LanternUtils.zeroFill(pw);
469             }
470         }
471     }

473     private static char [] readSettingsPasswordCLI() throws IOException {
474         if (LanternHub.settings().isLocalPasswordInitialized() == false) {
475             while (true) {
476                 // XXX i18n
477                 System.out.print("Please enter a password to protect your local data:");
478                 System.out.flush();
479                 final char [] pw1 = LanternUtils.readPasswordCLI();
480                 if (pw1.length == 0) {
481                     System.out.println("password cannot be blank, please try again.");
482                     System.out.flush();
483                     continue;
484                 }
485                 System.out.print("Please enter password again:");
486                 System.out.flush();
487                 final char [] pw2 = LanternUtils.readPasswordCLI();
488                 if (Arrays.equals(pw1, pw2)) {
489                     // zero out pw2
490                     LanternUtils.zeroFill(pw2);
491                     return pw1;
492                 }
493                 else {
494                     LanternUtils.zeroFill(pw1);
495                     LanternUtils.zeroFill(pw2);
496                     System.out.println("passwords did not match, please try again.");
497                     System.out.flush();
498                 }
499             }
500         }
501         else {
502             System.out.print("Please enter your lantern password:");
503             System.out.flush();
504             return LanternUtils.readPasswordCLI();
505         }
506     }

509     private static boolean unlockSettingsWithPassword(final char [] password)
510         throws GeneralSecurityException, IOException {
511         final boolean init = !LanternHub.settings().isLocalPasswordInitialized();
512         LanternHub.localCipherProvider().feedUserInput(password, init);
513         LanternHub.resetSettings(true);
514         final SettingsState.State ss = LanternHub.settings().getSettings().getState();
515         if (ss != SettingsState.State.SET) {
516             LOG.error("Settings did not unlock, state is {}", ss);
517             return false;
518         }
519         return true;
520     }

522     private static void loadLocalPasswordFile(final String pwFilename) {
523         final LocalCipherProvider lcp = LanternHub.localCipherProvider();
524         if (!lcp.requiresAdditionalUserInput()) {
525             LOG.error("Settings do not require a password to unlock.");
526             System.exit(1);
527         }
528 
529         if (StringUtils.isBlank(pwFilename)) {
530             LOG.error("No filename specified to --{}", OPTION_PASSWORD_FILE);
531             System.exit(1);
532         }
533         final File pwFile = new File(pwFilename);
534         if (!pwFile.exists() && pwFile.canRead()) {
535             LOG.error("Unable to read password from {}", pwFilename);
536             System.exit(1);
537         }
538 
539         LOG.info("Reading local password from file \"{}\"", pwFilename);
540         try {
541             final String pw = FileUtils.readLines(pwFile, "US-ASCII").get(0);
542             final boolean init = !LanternHub.settings().isLocalPasswordInitialized();
543             lcp.feedUserInput(pw.toCharArray(), init);
544         }
545         catch (final IndexOutOfBoundsException e) {
546             LOG.error("Password in file \"{}\" was incorrect", pwFilename);
547             System.exit(1);
548         }
549         catch (final InvalidKeyException e) {
550             LOG.error("Password in file \"{}\" was incorrect", pwFilename);
551             System.exit(1);
552         }
553         catch (final GeneralSecurityException e) {
554             LOG.error("Failed to initialize using password in file \"{}\": {}", pwFilename, e);
555             System.exit(1);
556         }
557         catch (final IOException e) {
558             LOG.error("Failed to initialize using password in file \"{}\": {}", pwFilename, e);
559             System.exit(1);
560         }        
561     }

580     public static void launchLantern() {
581         LOG.debug("Launching Lantern...");
582         final KeyStoreManager proxyKeyStore = LanternHub.getKeyStoreManager();
583         
584         final HttpRequestFilter publicOnlyRequestFilter = 
585             new PublicIpsOnlyRequestFilter();
586         
587         // Note that just passing in the keystore manager triggers this to 
588         // become an SSL proxy server.
589         final int staticRandomPort = LanternHub.settings().getServerPort();
590         final StatsTrackingDefaultHttpProxyServer sslProxy =
591             new StatsTrackingDefaultHttpProxyServer(staticRandomPort,
592             new HttpResponseFilters() {
593                 @Override
594                 public HttpFilter getFilter(String arg0) {
595                     return null;
596                 }
597             }, null, proxyKeyStore, publicOnlyRequestFilter);
598         LOG.debug("SSL port is {}", staticRandomPort);
599         //final org.littleshoot.proxy.HttpProxyServer sslProxy = 
600         //    new DefaultHttpProxyServer(LanternHub.randomSslPort());
601         sslProxy.start(false, false);
602          
603         // The reason this exists is complicated. It's for the case when the
604         // offerer gets an incoming connection from the answerer, and then
605         // only on the answerer side. The answerer "client" socket relays
606         // its data to the local proxy.
607         // See http://cdn.getlantern.org/IMAG0210.jpg
608         final org.littleshoot.proxy.HttpProxyServer plainTextProxy = 
609             new DefaultHttpProxyServer(
610                 LanternUtils.PLAINTEXT_LOCALHOST_PROXY_PORT,
611                 publicOnlyRequestFilter);
612         plainTextProxy.start(true, false);
613         
614         LOG.info("About to start Lantern server on port: "+
615             LanternConstants.LANTERN_LOCALHOST_HTTP_PORT);
616 
617 
618         /* delegate all calls to the current hub cookie tracker */
619         final CookieTracker hubTracker = new CookieTracker() {
620 
621             @Override
622             public void setCookies(Collection<Cookie> cookies, HttpRequest context) {
623                 LanternHub.cookieTracker().setCookies(cookies, context);
624             }
625 
626             @Override
627             public boolean wouldSendCookie(final Cookie cookie, final URI toRequestUri) {
628                 return LanternHub.cookieTracker().wouldSendCookie(cookie, toRequestUri);
629             }
630 
631             @Override
632             public boolean wouldSendCookie(final Cookie cookie, final URI toRequestUri, final boolean requireValueMatch) {
633                 return LanternHub.cookieTracker().wouldSendCookie(cookie, toRequestUri, requireValueMatch);
634             }
635 
636             @Override
637             public CookieFilter asOutboundCookieFilter(final HttpRequest request, final boolean requireValueMatch) throws URISyntaxException {

678         private void checkAutoConnect() {
679             if (done) {
680                 return;
681             }
682             if (LanternHub.settings().getSettings().getState() != SettingsState.State.SET) {
683                 LOG.info("not testing auto-connect, settings are not ready.");
684                 return;
685             }
686             
687             // only test once.
688             done = true;
689             
690             LOG.info("Settings loaded, testing auto-connect behavior");
691             // This won't connect in the case where the user hasn't entered 
692             // their user name and password and the user is running with a UI.
693             // Otherwise, it will connect.
694             final XmppHandler xmpp = LanternHub.xmppHandler();
695             if (LanternHub.settings().isConnectOnLaunch() &&
696                 (LanternUtils.isConfigured() || !LanternHub.settings().isUiEnabled())) {
697                 final Runnable runner = new Runnable() {
698                     @Override
699                     public void run() {
700                         try {
701                             xmpp.connect();
702                         } catch (final IOException e) {
703                             LOG.info("Could not login", e);
704                         } catch (final CredentialException e) {
705                             LOG.info("Bad credentials");
706                         }
707                     }
708                 };
709                 final Thread t = 
710                     new Thread(runner, "Auto-Connect-From-Settings-Ready");
711                 t.setDaemon(true);
712                 t.start();
713             } else {
714                 LOG.info("Not auto-logging in with settings:\n{}",
715                     LanternHub.settings());
716             }
717 
718             try {
719                 LanternHub.configurator().copyFireFoxExtension();
720             } catch (final IOException e) {
721                 LOG.error("Could not copy extension", e);
722             }
723         }

791     private static void handleError(final Throwable t, final boolean exit) {
792         LOG.error("Uncaught exception: "+t.getMessage(), t);
793         if (t instanceof SWTError || t.getMessage().contains("SWTError")) {
794             System.out.println(
795                 "To run without a UI, run lantern with the --" + 
796                 OPTION_DISABLE_UI +
797                 " command line argument");
798         } 
799         else if (!lanternStarted && LanternHub.settings().isUiEnabled()) {
800             LOG.info("Showing error to user...");
801             LanternHub.dashboard().showMessage("Startup Error",
802                "We're sorry, but there was an error starting Lantern " +
803                "described as '"+t.getMessage()+"'.");
804         }
805         if (exit) {
806             LOG.info("Exiting Lantern");
807             System.exit(1);
808         }
