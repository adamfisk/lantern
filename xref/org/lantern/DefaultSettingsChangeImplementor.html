41      @Override
42      public void setStartAtLogin(final boolean start) {
43          if (SystemUtils.IS_OS_MAC_OSX && this.launchdPlist.isFile()) {
44              log.info("Setting start at login to "+start);
45              LanternUtils.replaceInFile(this.launchdPlist, 
46                  "<"+!start+"/>", "<"+start+"/>");
47          } else if (SystemUtils.IS_OS_WINDOWS) {
48              final String key = 
49                  "HKCU\\Software\\Microsoft\\Windows\\CurrentVersion\\Run";
50              int result = 0;
51              if (start) {
52                  try {
53                      final String path = 
54                          "\"\\\""+new File("lantern.exe").getCanonicalPath()+"\\\"\"";
55                      result = WindowsRegistry.writeREG_SZ(key, "Lantern", path);
56                  } catch (final IOException e) {
57                      log.error("Could not get canonical path", e);
58                  }
59              } else {
60                  result = WindowsRegistry.writeREG_SZ(key, "Lantern", "");
61              }
62              
63              if (result != 0) {
64                  log.error("Error enabling proxy server? Result: "+result);
65              }
66          } else if (SystemUtils.IS_OS_LINUX) {
67              // TODO: Make this work on Linux!! 
68          }
69      }

71      @Override
72      public void setSystemProxy(final boolean isSystemProxy) {
73          if (isSystemProxy == LanternHub.settings().isSystemProxy()) {
74              log.info("System proxy setting is unchanged.");
75              return;
76          }
77          
78          log.info("Setting system proxy");
79          
80          // go ahead and change the setting so that it will affect
81          // shouldProxy. it will be set again by the api, but that
82          // doesn't matter.
83          LanternHub.settings().setSystemProxy(isSystemProxy);
84          if (!LanternHub.settings().isInitialSetupComplete()) {
85              return;
86          }
87          
88          final Runnable proxyRunner = new Runnable() {
89              @Override
90              public void run() {
91                  try {
92                      if (LanternUtils.shouldProxy() ) {
93                          Proxifier.startProxying();
94                      } else {
95                          Proxifier.stopProxying();
96                      }
97                  } catch (final Proxifier.ProxyConfigurationError e) {
98                      log.error("Proxy reconfiguration failed: {}", e);
99                  }
100             }
101         };
102         proxyQueue.execute(proxyRunner);
103     }

118     @Override
119     public void setEmail(final String email) {
120         final String storedEmail = LanternHub.settings().getEmail();
121         if ((email == null && storedEmail == null)) {
122             log.info("Email is unchanged.");
123             return;
124         }
125         if (storedEmail != null && storedEmail.equals(email)) {
126             log.info("Email is unchanged.");
127             return;
128         }
129         log.info("Email address changed. Clearing user specific settings");
130         LanternHub.resetUserConfig();

133     @Override
134     public void setGetMode(final boolean getMode) {
135         // When we move to give mode, we want to start advertising our 
136         // ID and to start accepting incoming connections.
137         
138         // We we move to get mode, we want to stop advertising our ID and to
139         // stop accepting incoming connections.
140 
141         if (getMode == LanternHub.settings().isGetMode()) {
142             log.info("Mode is unchanged.");
143             return;
144         }
145         if (!LanternUtils.isConfigured()) {
146             log.info("Not implementing mode change -- not configured.");
147             return;
148         }
149         
150         // Go ahead and set the setting although it will also be
151         // updated by the api as well.  We want to make sure the
152         // state seen by the following calls is consistent with
153         // this flag being aspirational vs. representational
154         LanternHub.settings().setGetMode(getMode);
155         
156         // We disconnect and reconnect to create a new Jabber ID that will 
157         // not advertise us as a connection point.
158         LanternHub.xmppHandler().disconnect();
159         try {
160             try {
161                 LanternHub.xmppHandler().connect();
162                 
163                 // TODO: This isn't quite right. We don't necessarily have
164                 // proxies to connect to at this point, and we shouldn't set
165                 // the OS proxy until we do.
166                 if (LanternHub.settings().isInitialSetupComplete()) {
167                     // may need to modify the proxying state
168                     if (LanternUtils.shouldProxy()) {
169                         Proxifier.startProxying();
170                     } else {
171                         Proxifier.stopProxying();
172                     }
173                 }
174             } catch (final IOException e) {
175                 log.info("Could not login", e);
176                 // Don't proxy if there's some error connecting.
177                 if (LanternHub.settings().isInitialSetupComplete()) {
178                     Proxifier.stopProxying();
179                 }
180             }
181         } catch (final Proxifier.ProxyConfigurationError e) {
182             log.info("Proxy auto-configuration failed: {}", e);
183         }
184     }
