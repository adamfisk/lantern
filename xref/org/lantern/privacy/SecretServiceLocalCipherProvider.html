71      private static void initDBUS() throws IOException {
72          synchronized(DBUS_INITIALIZED) {
73              final Boolean initialized = DBUS_INITIALIZED.get();
74              if (initialized == null || initialized.booleanValue() == false) {
75                  LanternUtils.loadJarLibrary(UnixIOException.class, "libunix-java.so");
76                  DBUS_INITIALIZED.set(Boolean.TRUE);
77              }

81      public static boolean secretServiceAvailable() {
82          DBusConnection conn = null;
83          try {
84              LOG.debug("Checking for Secret Service API availability...");
85              initDBUS();
86              conn = DBusConnection.getConnection(DBusConnection.SESSION);
87              final Service secretService = conn.getRemoteObject(
88                  BUS_NAME, SERVICE_PATH, Service.class);
89              final Pair<Variant, Path> result = secretService.OpenSession(ALGORITHM_PLAIN, new Variant(""));
90              if (NO_OBJECT.equals(result.b.getPath())) {
91                  LOG.info("Failed to negotiate plain session with Secret Service API.");
92                  return false;
93              }
94              else {
95                  final Session session = conn.getRemoteObject(
96                      BUS_NAME, result.b.getPath(), Session.class);
97                  session.Close();
98              }
99              LOG.debug("Found Secret Service API.");
100             return true;
101         } catch (Exception e) {
102             LOG.debug("Could not connect to Secret Service API at {}: {}", SERVICE_PATH, e);
103             return false;
104         } finally {
105             closeConnection(conn);
106         }

119     @Override
120     byte[] loadKeyData() throws IOException, GeneralSecurityException {
121         DBusConnection conn = null;
122         byte [] encodedKey = null;
123         try {
124             LOG.debug("Loading key data from Secret Service API...");
125             initDBUS();
126 
127             conn = DBusConnection.getConnection(DBusConnection.SESSION);
128             final Service secretService = conn.getRemoteObject(
129                 BUS_NAME, SERVICE_PATH, Service.class);
130 
131             // negotiate a "plain" session (no encryption)
132             final Pair<Variant, Path> result = secretService.OpenSession(ALGORITHM_PLAIN, new Variant(""));
133             if (NO_OBJECT.equals(result.b.getPath())) {
134                 throw new IOException("Unable to negotiate DBus session!");
135             }
136             final Path sessionPath = result.b;
137             final Session session = conn.getRemoteObject(BUS_NAME, sessionPath.getPath(), Session.class);
138 
139             LOG.debug("Requesting Item....");
140             final Map<String, String> secretAttrs = new HashMap<String, String>();
141             secretAttrs.put(SECRET_ATTR_NAME, 
142                             SECRET_ATTR_VALUE);
143             final Pair<List<Path>, List<Path>> items = secretService.SearchItems(secretAttrs);
144             LOG.debug("Got {} unlocked / {} locked secret items.", items.a.size(), items.b.size());
145 
146             Item secretItem = null;
147             if (items.a.isEmpty()) {
148                 if (items.b.isEmpty()) {
149                     LOG.debug("Secret item is missing!");
150                     throw new GeneralSecurityException("Unable to locate secret in keychain!");
151                 }
152                 else {
153                     LOG.debug("Secret is locked! Attempting to unlock...");
154                     final Path secretItemPath = items.b.get(0);
155                     if (unlockPath(secretItemPath, secretService, conn)) {
156                         LOG.debug("Assuming all is well using path {}...", secretItemPath.getPath());
157                         secretItem = conn.getRemoteObject(BUS_NAME, secretItemPath.getPath(), Item.class);
158                     }
159                     else {
160                         LOG.error("Failed to unlock lantern keychain item.");
161                         throw new GeneralSecurityException("Failed to unlock lantern keychain item.");
162                     }
163                 }
164             }
165             else {
166                 LOG.debug("Secret is unlocked...");
167                 final Path itemPath = items.a.get(0);
168                 secretItem = conn.getRemoteObject(BUS_NAME, itemPath.getPath(), Item.class);
169             }
170             final Secret secret = secretItem.GetSecret(sessionPath);
171             encodedKey = new byte[secret.value.size()];
172             int i = 0; 
173             for (final Byte b : secret.value) {
174                 encodedKey[i++] = b; 
175             }
176             session.Close();
177 
178             final Base64 base64 = new Base64();
179             return base64.decode(encodedKey);
180         } catch (DBusException e) {
181             throw new IOException(e);
182         } catch (InterruptedException e) {
183             throw new IOException(e);
184         }finally {
185             if (encodedKey != null) {
186                 Arrays.fill(encodedKey, (byte) 0);
187             }
188             closeConnection(conn);
189         }
190     }

192     @Override
193     void storeKeyData(final byte[] key) throws IOException, GeneralSecurityException {
194         byte [] encodedKey = null;
195         List<Byte> secretValue = null;
196          
197         DBusConnection conn = null;
198         try {
199             LOG.debug("Storing key data via Secret Service API...");
200             initDBUS();
201             conn = DBusConnection.getConnection(DBusConnection.SESSION);
202             final Service secretService = conn.getRemoteObject(
203                 BUS_NAME, SERVICE_PATH, Service.class);
204 
205             final Collection collection = conn.getRemoteObject(
206                 BUS_NAME, COLLECTION_PATH, Collection.class);
207 
208             // Collection properties do not seem to be implemented...
209             // So instead of checking, just request an unlock in all cases 
210             if (!unlockPath(new Path(COLLECTION_PATH), secretService, conn)) {
211                 throw new GeneralSecurityException("Unable to unlock secret collection.");   
212             }
213 
214             // negotiate a "plain" session (no encryption)
215             final Pair<Variant, Path> result = secretService.OpenSession(ALGORITHM_PLAIN, new Variant(""));
216             if (NO_OBJECT.equals(result.b.getPath())) {
217                 throw new IOException("Unable to negotiate DBus session!");
218             }
219             final Path sessionPath = result.b;
220             final Session session = conn.getRemoteObject(
221                 BUS_NAME, sessionPath.getPath(), Session.class);
222 
223             // construct a "Secret"
224             final Base64 base64 = new Base64();
225             encodedKey = base64.encode(key);
226             secretValue = new ArrayList<Byte>(encodedKey.length);
227             for (byte b : encodedKey) {
228                 secretValue.add(b);
229             }
230             final Secret secret = new Secret(sessionPath, new ArrayList<Byte>(), secretValue, SECRET_CONTENT_TYPE);
231             final Map<String, Variant> secretProps = new HashMap<String, Variant>();
232             secretProps.put(SECRET_LABEL_PROPERTY,
233                             new Variant(LANTERN_KEY_LABEL));
234 
235             // these are the "attributes" of the item, not to be confused 
236             // with properties... 
237             final Map<String, String> secretAttrs = new HashMap<String, String>();
238             secretAttrs.put(SECRET_ATTR_NAME, SECRET_ATTR_VALUE);
239             secretProps.put(SECRET_ATTRIBUTES_PROPERTY, new Variant(secretAttrs, "a{ss}"));
240 
241             // add the item to the collection...
242             LOG.debug("Requesting CreateItem....");
243             final Pair<Path, Path> createResult = collection.CreateItem(
244                 secretProps, secret, true);
245 
246             final Path itemPath = createResult.a;
247             final Path promptPath = createResult.b;
248 
249             // user may need to be prompted... for what is unclear.
250             if (!NO_OBJECT.equals(promptPath.getPath())) {
251                 Prompt.Completed sig = prompt(promptPath, conn);
252                 // there does not seem to be a specification of what this means in this case... ?
253                 // possibly the path to the item or / on failure?
254                 LOG.debug("Prompt completed with dismissed={} result={}", sig.dismissed, sig.result);
255                 final List<Path> createdItemPaths = (List<Path>) sig.result.getValue();
256                 Path createdItemPath = null;
257                 if (createdItemPaths.size() > 0 ) {
258                     createdItemPath = createdItemPaths.get(0);
259                 }
260                 if (sig.dismissed == true || createdItemPath == null || NO_OBJECT.equals(createdItemPath.getPath())) {
261                     LOG.error("Failed to create lantern keychain item.");
262                     throw new GeneralSecurityException("Failed to create lantern keychain item.");
263                 }
264             }
265             else {
266                 LOG.debug("No prompt requested.");
267             }
268 
269             session.Close();
270             LOG.debug("Finished storing key in Secret Service API...");
271         } catch (DBusException e) {
272             throw new IOException(e);
273         } catch (InterruptedException e) {
274             throw new IOException(e);
275         } finally {
276             if (encodedKey != null) {
277                 Arrays.fill(encodedKey, (byte) 0);
278             }
279             closeConnection(conn);
280         }

283     private boolean unlockPath(final Path path, final Service secretService, final DBusConnection conn)
284         throws DBusException, InterruptedException {
285         LOG.debug("Requesting unlock of path {}", path.getPath());
286         final List<Path> pathsToUnlock = new ArrayList<Path>(1);
287         pathsToUnlock.add(path);
288         final Pair<List<Path>, Path> unlockResult = secretService.Unlock(pathsToUnlock);
289         if (unlockResult.a.size() > 0) {
290             LOG.debug("Path unlocked without prompt...");
291             return true;
292         }
293         // if there is a prompt 
294         else if (!NO_OBJECT.equals(unlockResult.b.getPath())) {
295             LOG.debug("Prompting user to unlock....");
296             Prompt.Completed sig = prompt(unlockResult.b, conn);
297             LOG.debug("Prompt completed with dismissed={} result={}", sig.dismissed, sig.result);
298             final List<Path> unlockedPaths = (List<Path>) sig.result.getValue();
299             Path unlockedPath = null; 
300             if (unlockedPaths.size() > 0) {
301                 unlockedPath = unlockedPaths.get(0);
302             }
303             if (sig.dismissed == true || unlockedPath == null || NO_OBJECT.equals(unlockedPath.getPath())) {
304                 LOG.error("Failed to unlock path {}", path.getPath());
305                 return false;
306             }
307             else {
308                 return true;
309             }
310         }
311         else {
312             LOG.debug("Confused, no prompt and no unlocked objects, failing.");
313             return false;
314         }

317     private Prompt.Completed prompt(final Path promptPath, final DBusConnection conn)
318         throws InterruptedException, DBusException {
319         LOG.debug("Prompt required...");
320         final Prompt prompt = conn.getRemoteObject(
321             BUS_NAME, promptPath.getPath(), Prompt.class);
322 
323         final PromptHandler handler = new PromptHandler();
324         conn.addSigHandler(Prompt.Completed.class, prompt, handler);
325         LOG.debug("added signal handler, calling prompt");
326         try {
327             prompt.Prompt("0"); // XXX window-id ?
328             LOG.debug("Waiting for prompt to complete...");
329             return handler.await();
330         } finally {
331             conn.removeSigHandler(Prompt.Completed.class, prompt, handler);
332         }
333     }

335     /**
336      * This is a DBusSigHandler that waits for the 
337      * response to a user prompt by sitting on a 
338      * Condition variable. 
339      */
340     private class PromptHandler implements DBusSigHandler {
341 
342         private final Lock lock; 
343         private final Condition gotResult; 
344         private DBusSignal sig = null;
345 
346         public PromptHandler() {
347             lock = new ReentrantLock();
348             gotResult = lock.newCondition();
349         }
350 
351         @Override
352         public void handle(final DBusSignal sig) {
353             this.sig = sig;
354             lock.lock();
355             try {
356                 gotResult.signal();
357             } finally {
358                 lock.unlock();
359             }
360         }
361 
362         public Prompt.Completed await() throws InterruptedException {
363             lock.lock();
364             try {
365                 while (sig == null) {
366                     gotResult.await();
367                 }
368                 return (Prompt.Completed) sig;
369             } finally {
370                 lock.unlock();
371             }
372         }
