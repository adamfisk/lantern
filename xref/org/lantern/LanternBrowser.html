188             private void processEvent(final LocationEvent event) {
189                 final String location = event.location;
190                 log.info("Got location: {}", location);
191 
192                 if (location.endsWith("-copy.html")) {
193                     // This just means it's a request we've already prepared
194                     // for serving. If we don't do this check, we'll get an
195                     // infinite loop of copies.
196                     log.info("Accepting copied location");
197                     return;
198                 }
199                 // else if (location.contains("setLocalPassword0")) {
200                 // }
201                 else if (location.contains("setLocalPassword1")) {
202                     final String args = 
203                         StringUtils.substringAfter(location, "&");
204                     if (StringUtils.isBlank(args)) {
205                         log.error("Weird location: {}", location);
206                         return;
207                     }
208                     final String password1 = 
209                         StringUtils.substringBetween(location, "&password1=", "&");
210                     final String password2 = 
211                         StringUtils.substringAfter(location, "&password2=");
212                     if (StringUtils.isBlank(password1)) {
213                         startVals.put("error_message", "Password cannot be blank");
214                         // startVals.put("error_message", I18n.tr("Password cannot be blank"));

216                     }
217                     else if (!password1.equals(password2)) {
218                         startVals.put("error_message", "Passwords did not match");
219                         // startVals.put("error_message", I18n.tr("Passwords did not match"));
220                         setUrl(startFile, startVals);
221                     }
222                     else {
223                         passwordBuf.append(password1);
224                         close();
225                     }
226                 }
227                 event.doit = false;
228             }
229         });
230

234         shell.forceActive();
235         while (!shell.isDisposed()) {
236             if (!this.display.readAndDispatch())
237                 this.display.sleep();
238         }
239 
240         char[] passwordChars = new char[passwordBuf.length()];
241         passwordBuf.getChars(0, passwordBuf.length(), passwordChars, 0);

491                     // The user could be re-configuring their system. Make sure
492                     // force is no longer active.
