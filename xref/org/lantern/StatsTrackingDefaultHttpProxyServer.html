116     public StatsTrackingDefaultHttpProxyServer(final int port,
117         final HttpResponseFilters responseFilters,
118         final ChainProxyManager chainProxyManager, final KeyStoreManager ksm,
119         final HttpRequestFilter requestFilter) {
120         this.port = port;
121         this.responseFilters = responseFilters;
122         this.ksm = ksm;
123         this.requestFilter = requestFilter;
124         this.chainProxyManager = chainProxyManager;
125         Thread.setDefaultUncaughtExceptionHandler(new UncaughtExceptionHandler() {
126             @Override
127             public void uncaughtException(final Thread t, final Throwable e) {
128                 log.error("Uncaught throwable", e);
129             }
130         });
131 
132         this.serverBootstrap = new ServerBootstrap(
133             new NioServerSocketChannelFactory(
134                 Executors.newCachedThreadPool(),
135                 Executors.newCachedThreadPool()));
136     }

142     public void start(final boolean localOnly, final boolean anyAddress) {
143         log.info("Starting proxy on port: "+this.port);
144         final HttpServerPipelineFactory factory =
145             new StatsTrackingHttpServerPipelineFactory(authenticationManager,
146                 this.allChannels, this.chainProxyManager, this.ksm,
147                 new StatsTrackingDefaultRelayPipelineFactoryFactory(chainProxyManager,
148                     this.responseFilters, this.requestFilter,
149                     this.allChannels));
150         serverBootstrap.setPipelineFactory(factory);
151 
152         // Binding only to localhost can significantly improve the security of
153         // the proxy.
154         InetSocketAddress isa;
155         if (localOnly) {
156             isa = new InetSocketAddress("127.0.0.1", port);
157         }
158         else if (anyAddress) {
159             isa = new InetSocketAddress(port);
160         } else {
161             try {
162                 isa = new InetSocketAddress(NetworkUtils.getLocalHost(), port);
163             } catch (final UnknownHostException e) {
164                 log.error("Could not get local host?", e);
165                 isa = new InetSocketAddress(port);
166             }
167         }
168         final Channel channel = serverBootstrap.bind(isa);
169         allChannels.add(channel);
170 
171         Runtime.getRuntime().addShutdownHook(new Thread(new Runnable() {
172             @Override
173             public void run() {
174                 stop();
175             }
176         }));
177 
178         /*
179         final ServerBootstrap sslBootstrap = new ServerBootstrap(
180             new NioServerSocketChannelFactory(
181                 Executors.newCachedThreadPool(),
182                 Executors.newCachedThreadPool()));
183         sslBootstrap.setPipelineFactory(new HttpsServerPipelineFactory());
184         sslBootstrap.bind(new InetSocketAddress("127.0.0.1", 8443));
185         */
186     }

188     public void stop() {
189         log.info("Shutting down proxy");
190         final ChannelGroupFuture future = allChannels.close();
191         future.awaitUninterruptibly(6*1000);
192         serverBootstrap.releaseExternalResources();
193         log.info("Done shutting down proxy");
194     }

205     private static class StatsTrackingHttpServerPipelineFactory 
206         extends HttpServerPipelineFactory {
207         
208         public StatsTrackingHttpServerPipelineFactory(
209             final ProxyAuthorizationManager authorizationManager, 
210             final ChannelGroup channelGroup, 
211             final ChainProxyManager chainProxyManager, final KeyStoreManager ksm,
212             final RelayPipelineFactoryFactory relayPipelineFactoryFactory) {
213             super(authorizationManager, channelGroup, chainProxyManager, ksm, relayPipelineFactoryFactory);
214         }
215 
216         @Override
217         public ChannelPipeline getPipeline() throws Exception {
218             ChannelPipeline pipeline = super.getPipeline();
219             pipeline.addFirst("stats", new StatsTrackingHandler() {
220                 public void addUpBytes(long bytes, Channel channel) {
221                     statsTracker().addUpBytesToPeers(bytes, channel);
222                 }
223                 public void addDownBytes(long bytes, Channel channel) {
224                     statsTracker().addDownBytesFromPeers(bytes, channel);
225                 }
226             });
227             return pipeline;
228         }
229     }

234         public StatsTrackingDefaultRelayPipelineFactoryFactory(

236             final HttpResponseFilters responseFilters,

242         @Override
243         public ChannelPipelineFactory getRelayPipelineFactory(
244             final HttpRequest httpRequest,final Channel browserToProxyChannel,
245             final RelayListener relayListener) {
246             final ChannelPipelineFactory innerFactory =
247                     super.getRelayPipelineFactory(httpRequest, browserToProxyChannel, relayListener);
248 
249             return new ChannelPipelineFactory() {
250                 @Override
251                 public ChannelPipeline getPipeline() throws Exception {
252                     ChannelPipeline pipeline = innerFactory.getPipeline();
253                     pipeline.addFirst("stats", new StatsTrackingHandler() {
254                         @Override
255                         public void addUpBytes(final long bytes, final Channel channel) {
256                             statsTracker().addUpBytesForPeers(bytes, browserToProxyChannel);
257                         }
258                         @Override
259                         public void addDownBytes(final long bytes, final Channel channel) {
260                             statsTracker().addDownBytesForPeers(bytes, browserToProxyChannel);
261                         }
262                     });
263                     return pipeline;
264                 }
265             };
266         }
