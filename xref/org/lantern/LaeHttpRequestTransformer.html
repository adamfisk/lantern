20          
21          final String host = proxyAddress.getHostName();
22          final String proxyBaseUri = "https://" + host;
23          if (!uri.startsWith(proxyBaseUri)) {
24              request.setHeader("Host", host);
25              final String scheme = uri.substring(0, uri.indexOf(':'));
26              final String rest = uri.substring(scheme.length() + 3);
27              final String proxyUri = proxyBaseUri + "/" + scheme + "/" + rest;
28              log.debug("proxyUri: " + proxyUri);
29              request.setUri(proxyUri);
30          } else {
31              log.info("NOT MODIFYING URI -- ALREADY HAS FREELANTERN");
32          }
33      }
